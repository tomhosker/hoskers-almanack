\settowidth{\versewidth}{Breath this way and cool me,}
\begin{verse}[\versewidth]
\poemlines{5}
Thou joy of my life,\footnotetext{$\xi$ From `A Ballet', Sidney Godolphin (1610 -- 1643). \cite{obev}. The first Earl of Godolphin was a nephew of the poet. This poem is drawn from a somewhat longer one; these lines are the shepherd's speech to the nymph Amarillis.}\\* 
First love of my youth,\\
Thou safest of pleasures\\
And fullest of truth,\\
Thou purest of nymphs\\
And never more fair,\\
Breath this way and cool me,\\
Thou pitying air;\\
Come hither and hover\\
On every part,\\
Thou life of my sense\\*
And joy of my heart.
\end{verse}
