\settowidth{\versewidth}{Full fathom five thy father lies:}
\begin{verse}[\versewidth]
\poemlines{5}
Full fathom five thy father lies:\footnotetext{`A Sea Dirge', William Shakespeare (1564 -- 1616). \cite{treasury}. This song is sung by Ariel in \emph{The Tempest} I.2.}\\*
\vin Of his bones are coral made;\\
Those are pearls that were his eyes:\\
\vin Nothing of him that doth fade\\
But doth suffer a sea-change\\
Into something rich \& strange.\\
Sea-nymphs hourly ring his knell:\\
Hark! now I hear them --\\*
\vin \vin Ding-dong, bell.
\end{verse}
