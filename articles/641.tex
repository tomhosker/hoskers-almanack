\settowidth{\versewidth}{Sweet dreams of pleasant streams}
\begin{verse}[\versewidth]
\poemlines{5}
Sweet dreams form a shade\footnotetext{`Cradle Song', William Blake (1757 -- 1827). \cite{blakea}. The Almanackist has omitted the last three verses, being of inferior quality than the rest.}\\*
O'er my lovely infant's head.\\
Sweet dreams of pleasant streams\\*
By happy, silent, moony beams.\\!

Sweet sleep, with soft down\\*
Weave thy brows an infant crown.\\
Sweet sleep, angel mild,\\*
Hover o'er my happy child.\\!

Sweet smiles, in the night\\*
Hover over my delight.\\
Sweet smiles, mother's smile,\\*
All the livelong night beguile.\\!

Sweet moans, dovelike sighs,\\*
Chase not slumber from thine eyes.\\
Sweet moan, sweeter smiles:\\*
All the dovelike moans beguile.\\!

Sleep, sleep, happy child.\\*
(All creation slept \& smiled.)\\
Sleep, sleep, happy sleep,\\*
While o'er thee doth mother weep.
\end{verse}
