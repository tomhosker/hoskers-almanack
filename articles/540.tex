\settowidth{\versewidth}{Tell me not, sweet, I am unkind}
\begin{verse}[\versewidth]
\poemlines{5}
I prithee, let me bring thee where crabs grow;\footnotetext{William Shakespeare (1564 -- 1616). \cite{shakespeare}. These lines are uttered by Caliban in \emph{The Tempest} II.2.}\\*
And I with my long nails will dig thee pignuts,\\
Show thee a jay's nest, and instruct thee how\\
To snare the nimble marmoset. I'll bring thee\\
To clustering filberts, and sometimes I'll get thee\\*
Young scamels from the rock. Wilt thou go with me?
 \end{verse}
