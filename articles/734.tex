\settowidth{\versewidth}{\vin In the invisible storm}
\begin{verse}[\versewidth]
\poemlines{5}
O rose, thou art sick.\footnotetext{`The Sick Rose', William Blake (1757 -- 1827). \cite{blakea}.}\\*
\vin The invisible worm\\
That flies in the night\\*
\vin In the howling storm\\!

Has found out thy bed\\*
\vin Of crimson joy:\\
And his dark secret love\\*
\vin Does thy life destroy.
\end{verse}
